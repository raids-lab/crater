% Algorithm 4: 工作流 Checkpoint 与确认恢复 CheckpointResume
\begin{algorithm}[t]
\caption{工作流 Checkpoint 与确认恢复 CheckpointResume}
\label{alg:checkpoint-resume}
\DontPrintSemicolon
\KwData{当前 MASState $M$，确认工具调用 $\tau = (\text{name}, \text{args})$}
\KwResult{用户决策 $d \in \{\text{approve, modify, reject}\}$ 与恢复执行结果 $E$}

$\sigma \leftarrow \langle M.phase, M.plan, M.observation, M.action\_history, \tau \rangle$ \tcp*{冻结快照}
$\textsc{Persist}(\sigma)$ ; $cid \leftarrow \textsc{NewConfirmationID}()$ \;

\textsc{EmitEvent}(\texttt{confirmation\_required}, $\tau$, $cid$) \tcp*{SSE 推送前端}

$d \leftarrow \textsc{WaitForUserDecision}(cid, \text{timeout})$ \;

\uIf{$d = \text{reject}$}{
  $\textsc{AuditLog}(cid, \text{rejected})$ ;
  \Return $\text{(reject, }\varnothing\text{)}$ \;
}
\uElseIf{$d = \text{modify}$}{
  $\tau' \leftarrow \textsc{ApplyUserPatch}(\tau, d.\text{patch})$ \;
}
\Else{
  $\tau' \leftarrow \tau$ \;
}

$M \leftarrow \textsc{Restore}(\sigma)$ \tcp*{从快照恢复 MASState}
$E \leftarrow \textsc{Executor.Run}(\tau')$ ;
$M.action\_history \leftarrow M.action\_history \cup \{(\tau', E)\}$ \;

$\textsc{AuditLog}(cid, d, \tau', E)$ \;
\Return $(d, E)$
\end{algorithm}
