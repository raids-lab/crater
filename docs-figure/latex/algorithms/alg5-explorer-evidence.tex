% Algorithm 5: 探索器迭代式证据收集 ExplorerEvidence
\begin{algorithm}[t]
\caption{探索器迭代式证据收集 ExplorerEvidence}
\label{alg:explorer-evidence}
\DontPrintSemicolon
\KwData{规划产物 $\mathcal{P}$，只读工具集 $\mathcal{T}_{ro}$，最大迭代 $K$，无进展阈值 $\kappa$}
\KwResult{观察产物 $\mathcal{O}$ (ObservationArtifact)}

$\mathcal{O} \leftarrow \emptyset$ ;
$\mathcal{Q} \leftarrow \mathcal{P}.candidate\_tools$ \tcp*{候选工具队列}
$k \leftarrow 0$ ;
$no\_progress \leftarrow 0$ \;

\While{$k < K$ \textbf{and} $\mathcal{Q} \neq \varnothing$ \textbf{and} $no\_progress < \kappa$}{
  $\mathcal{T}_k \leftarrow \textsc{SelectNext}(\mathcal{Q}, \mathcal{O}; |\mathcal{T}_k| \leq 2)$ \tcp*{每轮 $\leq 2$ 个工具}

  \ForEach{$t \in \mathcal{T}_k$}{
    \If{$t \notin \mathcal{T}_{ro}$}{
      \textbf{continue} \tcp*{角色权限过滤}
    }
    $r_t \leftarrow \textsc{Invoke}(t, t.args)$ ;
    $\mathcal{O} \leftarrow \mathcal{O} \cup \{(t, r_t)\}$ \;
  }

  $\Delta \leftarrow \textsc{NoveltyScore}(\mathcal{O}, \mathcal{T}_k)$ \;
  $no\_progress \leftarrow no\_progress + 1$ \textbf{if } $\Delta < \epsilon$ \textbf{else } $0$ \;

  \If{\textsc{LLMSufficient}($\mathcal{O}$, $\mathcal{P}.goal$)}{
    \textbf{break} \tcp*{证据已充分}
  }

  $\mathcal{Q} \leftarrow \textsc{ReplanCandidates}(\mathcal{O}, \mathcal{T}_{ro})$ ;
  $k \leftarrow k + 1$ \;
}

$\mathcal{O}.summary \leftarrow \textsc{LLMSummarize}(\mathcal{O})$ \;
\Return $\mathcal{O}$
\end{algorithm}
