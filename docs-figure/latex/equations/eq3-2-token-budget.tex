% Equation 3.2: Token 预算按角色分配
B_{\text{role}} =
\begin{cases}
  \alpha_{P} \cdot B_{\text{total}}, & \text{role} = \text{Planner},\;\alpha_P \approx 0.20 \\[2pt]
  \alpha_{E} \cdot B_{\text{total}}, & \text{role} = \text{Explorer},\;\alpha_E \approx 0.45 \\[2pt]
  \alpha_{X} \cdot B_{\text{total}}, & \text{role} = \text{Executor},\;\alpha_X \approx 0.25 \\[2pt]
  \alpha_{V} \cdot B_{\text{total}}, & \text{role} = \text{Verifier},\;\alpha_V \approx 0.10 \\[2pt]
\end{cases}
\quad \text{满足}\;\;\sum_{r} \alpha_r = 1
